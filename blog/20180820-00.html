# coding: utf-8
def ysfh(m,n):
    # 二维数组 [1,0] - [100,0]
    s = [[i,0] for i in range(1,m+1)]
    # 从 1 开始
    k = 0
    # 直到只剩1个
    while(len(s) > 1):
        i_len = len(s)
        for i in range(n-1):
            # 经过 +1
            s[(k+i)%i_len][1] = s[(k+i)%i_len][1] + 1
        del(s[(k+n-1)%i_len])
        k = (k+n-1)%len(s)
        print(s)

if \'__main__\' == __name__:
    ysfh(100,6)
